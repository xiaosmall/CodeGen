1function [SlopeResult, PlantGain,ErrCode] = LinearSlope4Manual(f, r, n, MinFreqRange, MaxFreqRange)
 2%   Input parameters:
 3%
 4%   f - Frequencies vector. Assumed to have fixed length of 1000 (needs fixed length for codegen) 
 5%   r - Gains vector, in [dB]. As above.
 6%   n - Actual length of the f and r vectors.
 7%   MinFreqRange - Start frequency range.
 8%   MaxFreqRange - End frequency range.
 9%   Output parameters:
10%Output---------
11%   SlopeResult - the calculated slope for this frequency range.
12%   PlantGain - Calculated gain of plant.
13%   ErrCode --- if negative value, Daniel will display the ErrCode;
14%           --- [-3]--- input value out of range;
15%   All output parameters are 0 if the function fails to find a solution.
16%   plantGain calculation is based on each freq point(w) and G/(jw)^2,then make average of the whole interpolated freq points 
17%#codegen
18%init output
19SlopeResult = 0;
20PlantGain = 0;
21ErrCode  = 0;
22% var = f;
23% if isempty(var) || isnan(var) 
24%     ErrCode  = -3;
25% end
26ErrCode = inputVarCheck(f,-3);
27ErrCode = inputVarCheck(r,-3);
28ErrCode = inputVarCheck(n,-3);
29ErrCode = inputVarCheck(MinFreqRange,-3);
30ErrCode = inputVarCheck(MaxFreqRange,-3);
31
32if ErrCode <0
33    return;
34end
35
36df = 0.1;
37if MinFreqRange <= 0 | MaxFreqRange <=0
38    ErrCode = -4;
39end
40if ErrCode <0
41    return;
42end
43
44fnew = f(1):df:f(n); %% 1*XX 
45%%
46% nLength = length(fnew);
47% % fnew = fnew(:);%% do we need to XX*1 ? temp, will uncomment later;
48% % % // Den = reshape(Den,1,numel(Den));%%1*XX matrix transopse eg
49rdb = interp1(f(1:n),r(1:n),fnew);
50idxG = find(  fnew > MinFreqRange &  fnew < MaxFreqRange);
51if isempty(idxG)
52    ErrCode = -1;
53    return;
54end
55ActiveLength = length(idxG);
56if ActiveLength > length(fnew)
57    ErrCode = -2;
58    return;
59end
60xx= log10(fnew(idxG(1):idxG(ActiveLength)));
61
62[pCoefficient,~] = polyfit(xx,rdb(idxG(1):idxG(end)),1);
63SlopeResult = pCoefficient(1);
64%PlantGain22 = (2*pi)^2*10^(pCoefficient(2)/20)
65
66%%%for  plant gain calculation;
67gainLineRange = zeros(1000,1);
68rnondb = 10.^(rdb/20); %%non-dB
69
70for ii=1:ActiveLength
71      gainLineRange(ii) = abs(rnondb(idxG(ii)))*fnew(idxG(ii))*2*pi*fnew(idxG(ii))*2*pi;
72end
73vPlantGain  = sum(gainLineRange)/length(idxG); %%can not use mean of matlab for the zero value 
74PlantGain = vPlantGain(1);
